create table om_checkin(
	id int primary key,
    note nvarchar(255),
    created_date datetime not null,
    modified_date datetime not null,
    user_id int not null,
    foreign key(user_id) references om_user(id)
);
create table om_checkout(
	id int primary key,
    note nvarchar(255),
    created_date datetime not null,
    modified_date datetime not null,
    user_id int not null,
    foreign key(user_id) references om_user(id)
);
